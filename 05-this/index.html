myBind


Function.prototype.myBind = function(_this) {
    var fn = this;
    var args = [].slice.call(arguments, 1);

    return function () {
       var curArgs = [].slice.call(arguments);

       return fn.apply(_this, args.concat(curArgs));
    };
};
